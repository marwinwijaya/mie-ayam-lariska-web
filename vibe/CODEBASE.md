### FirebaseService (js/firebase-config.js)
**Purpose**: Firebase initialization and stock data management

**Public API**:
```javascript
window.FirebaseService = {
  config,               // Firebase config object
  db,                   // Firebase database reference
  stockRef,             // Stock data reference
  INITIAL_STOCK_DATA,   // Default stock data
  seedInitialStock(),   // Seed initial data
  onStockChange(),      // Listen to single item changes
  onAllStockChange(),   // Listen to all stock changes
  updateStock(),        // Update stock status
  updateStockWithRetry(), // Update with retry logic
  getStockStatus(),     // Get single item status
  getAllStock(),         // Get all stock data
  nameToStockId(),      // Convert name to slug
  isValidStockStatus()  // Validate status string
};
```

### StockService (js/stock-service.js)
**Purpose**: localStorage caching layer for offline resilience

**Public API**:
```javascript
window.StockService = {
  normalizeStockStatus(),       // Normalize status string
  saveStockData(),              // Save to localStorage
  getCachedStockData(),         // Read from localStorage
  clearStockCache(),            // Clear cache
  getStockStatus(),             // Get status with fallback
  mergeStockData(),             // Merge Firebase + cache
  processFirebaseData(),        // Normalize Firebase data
  processAndCacheFirebaseData(), // Process and save
  showErrorIndicator(),         // Show error banner
  hideErrorIndicator(),         // Hide error banner
  isUsingCachedData(),          // Check cache state
  setUsingCachedData()          // Set cache state
};
```
